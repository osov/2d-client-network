package protocol

type cs_connect struct{
	id_session text
}

type sc_close struct{}

type cs_ping struct{
	client_time uint64
}

type sc_pong struct{
	client_time uint64
	offset_time uint32
}

type sc_timestamp struct{
	offset_time uint32
}

type sc_init struct{
	server_start_time uint64
	offset_time uint32
	id_user uint32
	data text
}

type Vector2Int16 struct{
	x int16
	y int16
}

type Vector2Uint8 struct{
	x uint8
	y uint8
}

type EntityInfo struct{
	id uint16
	position Vector2Int16
	velocity Vector2Uint8
	angle uint8
}

type sc_world_state_update struct{
	list []EntityInfo
}

type sc_remove_e struct{
	id_entity uint16
}

type cs_input struct{
	key uint8
	state uint8
}

type cs_mouse_angle struct{
	angle uint8
}

type sc_join struct{
	id_user uint32
	id_entity uint16
}

type sc_leave struct{
	id_user uint32
	id uint16
}

type EntityBase struct{
	id uint16
	position Vector2Int16
}

type EntityBullet struct{
	id uint16
	position Vector2Int16
	velocity Vector2Uint8
	angle uint8
	offset_time uint32
}
